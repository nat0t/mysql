Подсчитать общее количество лайков десяти самым молодым пользователям (сколько лайков получили 10 самых молодых пользователей).

SELECT
  target_id,
  TIMESTAMPDIFF(
    YEAR,
    (SELECT birthday FROM profiles WHERE user_id = likes.target_id),
    NOW()) as age,
  COUNT(
  TIMESTAMPDIFF(
    YEAR,
    (SELECT birthday FROM profiles WHERE user_id = likes.target_id),
    NOW())) AS total
FROM likes
GROUP BY target_id
ORDER BY age
LIMIT 10;
