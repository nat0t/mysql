Определить кто больше поставил лайков (всего) - мужчины или женщины?

SELECT CASE
  WHEN
    (SELECT COUNT(
      (SELECT gender FROM profiles WHERE user_id = likes.user_id AND gender = 'f')) AS total_f
    FROM likes) >
    (SELECT COUNT(
      (SELECT gender FROM profiles WHERE user_id = likes.user_id AND gender = 'm')) AS total_m
    FROM likes)
  THEN
    'Женщин, поставивших лайки, больше, чем мужчин.'
  WHEN
    (SELECT COUNT(
      (SELECT gender FROM profiles WHERE user_id = likes.user_id AND gender = 'f')) AS total_f
    FROM likes) <
    (SELECT COUNT(
      (SELECT gender FROM profiles WHERE user_id = likes.user_id AND gender = 'm')) AS total_m
    FROM likes)
  THEN
    'Мужчин, поставивших лайки, больше, чем женщин.'
  ELSE
    'Лайки поставили одинаковое количество мужчин и женщин.'
END;
